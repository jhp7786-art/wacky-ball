    class SoundSynth {
      constructor() {
        this.ctx = null;
        this.muted = false;
        this.stretchOsc = null;
        this.stretchGain = null;
      }

      init() {
        if (this.ctx) return;
        this.ctx = new (window.AudioContext || window.webkitAudioContext)();
      }

      toggleMute() {
        this.muted = !this.muted;
        
        
        if (this.muted) this.stopStretch();
        return this.muted;
      }

      playStretch(pct) {
        if (this.muted) return;
        this.init();
        if (this.ctx.state === "suspended") this.ctx.resume();
        const freq = 120 + pct * 240;
        const vol  = 0.05 + pct * 0.12;
        if (!this.stretchOsc) {
          this.stretchOsc  = this.ctx.createOscillator();
          this.stretchOsc.type = "triangle";
          this.stretchGain = this.ctx.createGain();
          this.stretchOsc.connect(this.stretchGain);
          this.stretchGain.connect(this.ctx.destination);
          this.stretchOsc.frequency.setValueAtTime(freq, this.ctx.currentTime);
          this.stretchGain.gain.setValueAtTime(vol, this.ctx.currentTime);
          this.stretchOsc.start();
        } else {
          this.stretchOsc.frequency.setTargetAtTime(freq, this.ctx.currentTime, 0.05);
          this.stretchGain.gain.setTargetAtTime(vol, this.ctx.currentTime, 0.05);
        }
      }

      stopStretch() {
        if (this.stretchOsc) {
          try {
            const osc  = this.stretchOsc;
            const gain = this.stretchGain;
            this.stretchOsc  = null;
            this.stretchGain = null;
            gain.gain.setTargetAtTime(0, this.ctx.currentTime, 0.05);
            setTimeout(() => { osc.stop(); osc.disconnect(); gain.disconnect(); }, 100);
          } catch (e) { /* ignore */ }
        }
      }

      playLaunch() {
        if (this.muted) return;
        this.init();
        const now  = this.ctx.currentTime;
        const osc  = this.ctx.createOscillator();
        const gain = this.ctx.createGain();
        osc.type = "sine";
        osc.frequency.setValueAtTime(140, now);
        osc.frequency.exponentialRampToValueAtTime(480, now + 0.08);
        osc.frequency.exponentialRampToValueAtTime(70,  now + 0.18);
        gain.gain.setValueAtTime(0.35, now);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.18);
        osc.connect(gain); gain.connect(this.ctx.destination);
        osc.start(now); osc.stop(now + 0.2);
      }

      playBounce(isWall = true) {
        if (this.muted) return;
        this.init();
        const now      = this.ctx.currentTime;
        const osc      = this.ctx.createOscillator();
        const gain     = this.ctx.createGain();
        const baseFreq = isWall ? 300 : 170;
        osc.type = "triangle";
        osc.frequency.setValueAtTime(baseFreq, now);
        osc.frequency.exponentialRampToValueAtTime(baseFreq * 0.5, now + 0.09);
        gain.gain.setValueAtTime(0.12, now);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.09);
        osc.connect(gain); gain.connect(this.ctx.destination);
        osc.start(now); osc.stop(now + 0.1);
      }

      playRim() {
        if (this.muted) return;
        this.init();
        const now  = this.ctx.currentTime;
        const osc  = this.ctx.createOscillator();
        const gain = this.ctx.createGain();
        osc.type = "sine";
        osc.frequency.setValueAtTime(950, now);
        osc.frequency.setValueAtTime(750, now + 0.02);
        gain.gain.setValueAtTime(0.18, now);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.12);
        osc.connect(gain); gain.connect(this.ctx.destination);
        osc.start(now); osc.stop(now + 0.14);
      }

      playScore() {
        if (this.muted) return;
        this.init();
        const now   = this.ctx.currentTime;
        const notes = [523.25, 659.25, 783.99, 1046.50]; // C5-E5-G5-C6 arpeggio
        notes.forEach((freq, idx) => {
          const t    = now + idx * 0.08;
          const osc  = this.ctx.createOscillator();
          const gain = this.ctx.createGain();
          osc.type = "sine";
          osc.frequency.setValueAtTime(freq, t);
          gain.gain.setValueAtTime(0, t);
          gain.gain.linearRampToValueAtTime(0.18, t + 0.02);
          gain.gain.exponentialRampToValueAtTime(0.001, t + 0.28);
          osc.connect(gain); gain.connect(this.ctx.destination);
          osc.start(t); osc.stop(t + 0.32);
        });
      }

      playWin() {
        if (this.muted) return;
        this.init();
        const now   = this.ctx.currentTime;
        const notes = [523.25, 659.25, 783.99, 1046.50, 783.99, 1046.50, 1318.51];
        notes.forEach((freq, idx) => {
          const t      = now + idx * 0.07;
          const osc    = this.ctx.createOscillator();
          const gain   = this.ctx.createGain();
          const filter = this.ctx.createBiquadFilter();
          osc.type = "sawtooth";
          osc.frequency.setValueAtTime(freq, t);
          filter.type = "lowpass";
          filter.frequency.setValueAtTime(2000, t);
          gain.gain.setValueAtTime(0, t);
          gain.gain.linearRampToValueAtTime(0.12, t + 0.015);
          gain.gain.exponentialRampToValueAtTime(0.001, t + 0.24);
          osc.connect(filter); filter.connect(gain); gain.connect(this.ctx.destination);
          osc.start(t); osc.stop(t + 0.28);
        });
      }

      playClick() {
        if (this.muted) return;
        this.init();
        const now  = this.ctx.currentTime;
        const osc  = this.ctx.createOscillator();
        const gain = this.ctx.createGain();
        osc.type = "sine";
        osc.frequency.setValueAtTime(800, now);
        osc.frequency.exponentialRampToValueAtTime(200, now + 0.04);
        gain.gain.setValueAtTime(0.08, now);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.04);
        osc.connect(gain); gain.connect(this.ctx.destination);
        osc.start(now); osc.stop(now + 0.05);
      }

      // Paper crinkle swap sound
      playDiscard() {
        if (this.muted) return;
        this.init();
        const now        = this.ctx.currentTime;
        const bufferSize = this.ctx.sampleRate * 0.12;
        const buffer     = this.ctx.createBuffer(1, bufferSize, this.ctx.sampleRate);
        const data       = buffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) data[i] = Math.random() * 2 - 1;
        const noise  = this.ctx.createBufferSource();
        noise.buffer = buffer;
        const filter = this.ctx.createBiquadFilter();
        filter.type = "bandpass";
        filter.frequency.setValueAtTime(1200, now);
        filter.frequency.exponentialRampToValueAtTime(320, now + 0.11);
        const gain = this.ctx.createGain();
        gain.gain.setValueAtTime(0.2, now);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.11);
        noise.connect(filter); filter.connect(gain); gain.connect(this.ctx.destination);
        noise.start(now);
      }

      // Detuned dual-oscillator alarm buzzer
      playBuzzer() {
        if (this.muted) return;
        this.init();
        const now  = this.ctx.currentTime;
        const osc1 = this.ctx.createOscillator();
        const osc2 = this.ctx.createOscillator();
        const gain = this.ctx.createGain();
        osc1.type = "sawtooth"; osc1.frequency.setValueAtTime(110, now);
        osc2.type = "sawtooth"; osc2.frequency.setValueAtTime(114, now);
        gain.gain.setValueAtTime(0.35, now);
        gain.gain.linearRampToValueAtTime(0.35, now + 0.35);
        gain.gain.exponentialRampToValueAtTime(0.001, now + 0.48);
        osc1.connect(gain); osc2.connect(gain); gain.connect(this.ctx.destination);
        osc1.start(now); osc2.start(now);
        osc1.stop(now + 0.5); osc2.stop(now + 0.5);
      }

      // Squeaky squeegee wipe sound
      playWipe() {
        if (this.muted) return;
        this.init();
        const now     = this.ctx.currentTime;
        const osc     = this.ctx.createOscillator();
        const oscGain = this.ctx.createGain();
        osc.type = "sine";
        osc.frequency.setValueAtTime(600, now);
        osc.frequency.linearRampToValueAtTime(900, now + 0.08);
        osc.frequency.linearRampToValueAtTime(500, now + 0.2);
        oscGain.gain.setValueAtTime(0, now);
        oscGain.gain.linearRampToValueAtTime(0.15, now + 0.04);
        oscGain.gain.exponentialRampToValueAtTime(0.001, now + 0.2);
        osc.connect(oscGain); oscGain.connect(this.ctx.destination);
        osc.start(now); osc.stop(now + 0.22);
      }
    }

    function unlockSpeechAPI() {
      if (!window.speechSynthesis) return;
      try {
        const u = new SpeechSynthesisUtterance(" ");
        u.volume = 0;
        window.speechSynthesis.speak(u);
        window.removeEventListener('touchstart', unlockSpeechAPI);
        window.removeEventListener('click', unlockSpeechAPI);
      } catch (e) { console.warn("Speech unlock failed:", e); }
    }

    let currentUtterance = null; // Prevent GC of utterance

    function speakWord(word) {
      if (!word || !window.speechSynthesis) {
        console.warn("TTS Aborted: No word provided or speechSynthesis unsupported.");
        return;
      }
      try {
        console.log(`TTS: Canceling previous queue and preparing to speak: "${word}"`);
        window.speechSynthesis.cancel();
        
        currentUtterance = new SpeechSynthesisUtterance(`Spell the word: ${word}`);
        currentUtterance.rate  = 0.85; // kid-friendly speed
        
        if (window.speechSynthesis.getVoices) {
          const voices  = window.speechSynthesis.getVoices();
          const enVoice = voices.find(v => v.lang.startsWith("en") && v.localService) ||
                          voices.find(v => v.lang.startsWith("en"));
          if (enVoice) currentUtterance.voice = enVoice;
        }

        currentUtterance.onstart = (e) => {
          console.log(`TTS Event: [onstart] successfully started speaking "${word}"`);
        };
        currentUtterance.onend = (e) => {
          console.log(`TTS Event: [onend] finished speaking "${word}" after ${e.elapsedTime}ms`);
        };
        currentUtterance.onerror = (e) => {
          console.error(`TTS Event: [onerror] failed to speak "${word}". Error: ${e.error}`);
        };

        window.speechSynthesis.speak(currentUtterance);
        console.log(`TTS: Added utterance to synthesis queue.`);
      } catch (e) { 
        console.error("Speech speak function crashed:", e); 
      }
    }

    function loadGame() {
      if (currentProfile && currentProfile !== 'Guest') {
        try {
          const saved = localStorage.getItem('wackyball_' + currentProfile);
          if (saved) {
            const parsed = JSON.parse(saved);

    /** Pick a random unused word for the current level; triggers level-win if pool is empty. */
    function selectRandomWord() {
      const words     = LEVEL_WORDS[gameState.currentLevelIndex + 1];
      const available = words.filter(w => !gameState.completedWords.includes(w));
      if (available.length === 0) { triggerLevelWin(); return null; }
      gameState.currentWord = available[Math.floor(rand(0, available.length))];
      // Autoplay speech removed; user taps speaker icon to hear.
      return gameState.currentWord;
    }

    /** Assign the correct next letter to one random hoop; fill others with decoy letters. */
    function assignHoopLetters() {
      if (gameState.currentLetterIndex >= gameState.currentWord.length) return;
      const correctLetter  = gameState.currentWord[gameState.currentLetterIndex];
      const correctHoopIdx = Math.floor(rand(0, hoops.length));
      const alphabet       = "ABCDEFGHIJKLMNOPQRSTUVWXYZ";
      hoops.forEach((hoop, idx) => {
        if (idx === correctHoopIdx) {
          hoop.letter = correctLetter;
        } else {
          let randomL;
          do { randomL = alphabet[Math.floor(rand(0, alphabet.length))]; }
          while (randomL === correctLetter);
          hoop.letter = randomL;
        }
      });
    }

    /**
     * Shared vector-reflection used by both the obstacle and trash-can colliders.
     * Reflects ball velocity off the surface normal and nudges ball outside the collider.
     */
    function reflectBall(b, colliderPos, pushRadius) {
      const normal   = b.pos.sub(colliderPos).unit();
      const velocity = vec2(b.vx, b.vy);
      const dotVal   = velocity.dot(normal);
      if (dotVal < 0) {
        const reflected = velocity.sub(normal.scale(2 * dotVal)).scale(b.bounce);
        b.vx  = reflected.x;
        b.vy  = reflected.y;
        b.pos = colliderPos.add(normal.scale(pushRadius));
      }
    }

    function addConfetti(opt = {}) {
      for (let i = 0; i < 40; i++) {
        const p = add([
          pos(opt.pos ?? vec2(0,0)),
          rect(rand(6,12), rand(6,12), { radius:2 }),
          color(hsl2rgb(rand(0,1), 0.8, 0.6)),
          opacity(1), lifespan(2.5), rotate(rand(0,360)), anchor("center"), z(20)
        ]);
        const gravity = 700, drag = 0.92;
        const head = rand(-35,35) - 90, rad = (head*Math.PI)/180, spd = rand(300,700);
        let vx = Math.cos(rad)*spd, vy = Math.sin(rad)*spd;
        const spin = rand(-150,150);
        p.onUpdate(() => {
          vy += gravity*dt(); p.pos.x += vx*dt(); p.pos.y += vy*dt();
          p.angle += spin*dt(); p.opacity -= 0.45*dt(); vx *= drag; vy *= drag;
        });
      }
    }

      function clampPull(cur) {
        const dx=cur.x-dragStart.x, dy=cur.y-dragStart.y, dist=Math.sqrt(dx*dx+dy*dy);
        if (dist>DRAG_MAX) { const r=DRAG_MAX/dist; return vec2(dragStart.x+dx*r,dragStart.y+dy*r); }
        return vec2(cur.x,cur.y);
      }

      function isHoveringBatter(p) {
        return p.x>=BATTER_X-BATTER_W/2-15 && p.x<=BATTER_X+BATTER_W/2+15 &&
               p.y>=BATTER_Y-BATTER_H/2-15 && p.y<=BATTER_Y+BATTER_H/2+15;
      }

      function spawnBall(startPos, vx, vy) {
        if (ball) destroy(ball);
        inFlight = true;
        batter.scale = vec2(0.7,1.4); batter.gaspTimer = 0.6;
        const proj = gameState.currentProjectileType;

        // Per-type trail colour map
        const TRAIL_COLORS = {
          watermelon:[74,180,74], chicken:[255,220,50], gum:[255,120,180],
          beachball:[200,200,250], bowling:[80,80,90], anvil:[110,115,130],
        };

        ball = add([
          circle(BALL_R), pos(startPos), color(255,255,255), area(), z(4),
          {
            vx, vy, type:proj.type, isGood:proj.isGood, mass:proj.mass, bounce:proj.bounce, lastTrailTime:0,
            draw() { drawProjectile(this.type,vec2(0,0),BALL_R,1.0); },
            update() {
              if (isMenuPaused) return;
              if (config.wind) this.vx += curWindVal*dt();
              this.vx *= 1-0.012*dt()*60; // air resistance
              this.vy += GRAVITY*this.mass*dt();
              this.pos.x += this.vx*dt(); this.pos.y += this.vy*dt();

              // Colour-matched trail
              if (time()-this.lastTrailTime>0.035) {
                this.lastTrailTime=time();
                const tc=TRAIL_COLORS[this.type]||PAL.ball;
                add([circle(BALL_R-3),pos(this.pos),color(...tc),opacity(0.35),lifespan(0.25),anchor("center"),z(3),
                  {update(){this.opacity=Math.max(0,this.opacity-dt()*1.5);}}]);
              }

              // Wall / ceiling boundaries
              let hitWall=false;
              if(this.pos.x-BALL_R<0){ this.pos.x=BALL_R;    if(this.isGood){this.vx*=-this.bounce;sfx.playBounce(true);}else hitWall=true; }
              if(this.pos.x+BALL_R>W){ this.pos.x=W-BALL_R;  if(this.isGood){this.vx*=-this.bounce;sfx.playBounce(true);}else hitWall=true; }
              if(this.pos.y-BALL_R<0){ this.pos.y=BALL_R;    if(this.isGood){this.vy*=-this.bounce;sfx.playBounce(true);}else hitWall=true; }
              if (hitWall) {
                if(this.type==="mudpie") triggerMudSplatter(); else triggerStrike();
                destroy(this); ball=null; inFlight=false; wait(0.6,loadRandomProjectile); return;
              }

              // Bottom boundary (miss)
              if (this.pos.y>H+40) {
                inFlight=false; destroy(this); ball=null;
                batter.sadTimer=1.2; loadRandomProjectile();
              }

              checkHoops();
            }
          },
          "ball"
        ]);
      }

      function checkHoops() {
        if (!ball || !inFlight || isMenuPaused) return;

        if (!ball.isGood) {
          for (const h of hoops) {
            const dx=ball.pos.x-h.pos.x, dy=ball.pos.y-h.pos.y;
            if (Math.abs(dx)<HOOP_W/2+BALL_R && Math.abs(dy)<HOOP_H/2+BALL_R) {
              sfx.playRim();
              if(ball.type==="mudpie") triggerMudSplatter(); else triggerStrike();
              destroy(ball); ball=null; inFlight=false; wait(0.6,loadRandomProjectile); return;
            }
          }
          return;
        }

        for (const h of hoops) {
          if (h.stained) {
            if (ball.type==="soap") {
              const dx=ball.pos.x-h.pos.x, dy=ball.pos.y-h.pos.y;
              if (Math.abs(dx)<HOOP_W/2-2 && Math.abs(dy)<HOOP_H/2+3) {
                h.stained=false; h.netRef.flexY=24; h.netRef.flexSpeed=160;
                sfx.playScore(); spawnBubbleSparks(h.pos.x,h.pos.y);
                destroy(ball); inFlight=false; wait(0.6,loadRandomProjectile); return;
              }
            }
            continue; // stained hoops skip normal scoring
          }
          if (h.scored) continue;

          const dx=ball.pos.x-h.pos.x, dy=ball.pos.y-h.pos.y;
          if (Math.abs(dx)<HOOP_W/2-2 && Math.abs(dy)<HOOP_H/2+3) {
            h.scored=true; h.netRef.flexY=24; h.netRef.flexSpeed=160;
            h.opacity=0.9; wait(0.2,()=>{h.opacity=0.24;});
            sfx.playScore(); spawnSparks(h.pos.x,h.pos.y);
            if (ball.type==="watermelon") { spawnWatermelonSplat(h.pos.x,h.pos.y); h.stained=true; }
            if (h.letter===gameState.currentWord[gameState.currentLetterIndex]) {
              gameState.totalGameScore+=25; gameState.levelBasketsScored++; gameState.totalBasketsScored++;
              lightUpNextLetter();
            } else {
              triggerStrike();
            }
          }
        }
      }

      function lightUpNextLetter() {
        if (gameState.currentLetterIndex>=gameState.currentWord.length) return;
        const bubble=letterBubbles[gameState.currentLetterIndex];
        bubble.txt.text=bubble.letter;
        bubble.bg.color=rgb(...PAL.star); bubble.bg.opacity=1.0;
        bubble.txt.color=rgb(12,12,28);
        bubble.outlineCircle.color=rgb(255,255,255); bubble.outlineCircle.opacity=0.8;
        bubble.bg.scale=vec2(1.6); bubble.txt.scale=vec2(1.6);
        bubble.bg.onUpdate(()=>{bubble.bg.scale=bubble.bg.scale.lerp(vec2(1),dt()*10);});
        bubble.txt.onUpdate(()=>{bubble.txt.scale=bubble.txt.scale.lerp(vec2(1),dt()*10);});
        spawnSparks(bubble.bg.pos.x,bubble.bg.pos.y);
        batter.happyTimer=0.85;
        gameState.currentLetterIndex++;
        if (gameState.currentLetterIndex===gameState.currentWord.length) { wait(0.5,triggerWordWin); }
        else { assignHoopLetters(); }
      }

    function createRichButton(txt, p, onClickFn, colorBase = [50, 100, 200]) {
      const w = 260; const h = 60;
      // Shadow
      const shadow = add([rect(w, h, { radius: 12 }), pos(p.x, p.y + 6), anchor("center"), color(0, 0, 0), opacity(0.3), z(10)]);
      // Button
      const btn = add([rect(w, h, { radius: 12 }), pos(p.x, p.y), anchor("center"), color(...colorBase), area({cursor:"pointer"}), z(11)]);
      // Highlight/gradient illusion
      const hl = btn.add([rect(w - 6, h/2 - 4, { radius: 8 }), pos(0, -h/4 + 2), anchor("center"), color(255, 255, 255), opacity(0.15)]);
      // Text
      const label = btn.add([text(txt, { size: 20, font: "monospace", transform: (idx) => ({ color: rgb(255,255,255) }) }), anchor("center")]);

      btn.onHoverUpdate(() => {
        btn.scale = vec2(1.05); shadow.scale = vec2(1.05);
        btn.color = rgb(colorBase[0]+30, colorBase[1]+30, colorBase[2]+30);
      });
      btn.onHoverEnd(() => {
        btn.scale = vec2(1); shadow.scale = vec2(1);
        btn.color = rgb(...colorBase);
      });
      btn.onClick(() => {
        sfx.playClick();
        btn.pos.y += 4; shadow.pos.y -= 2;
        wait(0.1, () => {
          btn.pos.y -= 4; shadow.pos.y += 2;
          onClickFn();
        });
      });
      return btn;
    }

            function createShopList(title, items, type, startY) {
        add([text(title, { size: 20, font: "monospace" }), pos(W/2, startY), anchor("center"), color(150, 200, 255), z(5)]);
        items.forEach((item, i) => {
          const col = i % 2;

          const x = col === 0 ? W/2 - 115 : W/2 + 115;
          const y = startY + 40 + rowIdx * 45;
          const unlocked = (type === "skin" && gameState.unlockedPlayerSkins.includes(item.id)) ||
                           (type === "ball" && gameState.unlockedBallColors.includes(item.id)) ||
                           (type === "bg" && gameState.unlockedBackgrounds.includes(item.id));
          const equipped = (type === "skin" && gameState.equippedPlayerSkin === item.id) ||
                           (type === "ball" && gameState.equippedBallColor === item.id) ||
                           (type === "bg" && gameState.equippedBackground === item.id);
                           
          let btnColor = unlocked ? (equipped ? [50, 200, 50] : [100, 100, 100]) : [200, 150, 50];
          let statusTxt = equipped ? "EQP" : (unlocked ? "OWN" : `${item.cost}c`);
          
          const row = add([rect(210, 36, { radius: 6 }), pos(x, y), anchor("center"), color(...btnColor), area({cursor:"pointer"}), z(5)]);
          row.add([text(item.name, { size: 12, font: "monospace" }), pos(-95, 0), anchor("left"), color(255,255,255)]);

              sfx.playClick();
              gameState.coins -= item.cost;

              if (type === "ball") gameState.unlockedBallColors.push(item.id);
              if (type === "bg") gameState.unlockedBackgrounds.push(item.id);

              sfx.playBuzzer();
              row.use(color(200, 50, 50));
              wait(0.2, () => row.use(color(...btnColor)));
            }
          });
        });
